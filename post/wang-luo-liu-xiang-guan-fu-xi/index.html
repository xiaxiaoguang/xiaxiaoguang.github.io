inline int dfs(int u,int nflow)
{
	if(nflow==0||u==t)return nflow;
	int ret=nflow,a=0;
	for(int &i=cur[u]; i; i=nxt[i]) {
		int v=to[i];
		if(h[v]==h[u]+1&&(a=dfs(v,std::min(nflow,flow[i])))) {
			flow[i]-=a;
			flow[i^1]+=a;
			ret-=a;
			if(!ret)break;
		}
	}
	if(ret==nflow)h[u]=-1;
	return (nflow-ret);
}

int mrk[MAXN];
inline int dfs(int u,int nflow) {
	if(u==t) {
		maxcost+=dis[u]*nflow;
		return nflow;
	}
	mrk[u]=1;
	int ret=0;
	for(int &i=cur[u]; i; i=nxt[i]) {
		int v=to[i];
		if(!mrk[v]&&abs(dis[v]-dis[u]-w[i])<=eps&&flow[i]) {
			int a=dfs(v,min(nflow-ret,flow[i]));
			if(a) {
				flow[i]-=a;
				flow[i^1]+=a;
				ret+=a;
				if(ret==nflow)break;
			}
		}
	}
	return (ret);
}
