#include<iostream>
#include<cstdio>
#include<cstring>
#define ll long long
using namespace std;
ll a, b, c, d;
ll ans1, ans2;
int main() {
	scanf("%lld%lld%lld%lld", &a, &b, &c, &d);
	if(a == 0 && b == 0)return puts("0"), 0;
	if(a == 0 || b == 0) {
		return printf("%lld\n", d), 0;
	}
	if(a == b) {
		ans1 = c;
		ans2 = 2 * d;
		printf("%lld\n", min(ans1, ans2));
	} else {
		ans1 = c + d;
		ans2 = 2 * d;
		printf("%lld\n", min(ans1, ans2));
	}
	return 0;
}
