	ll tr[MAXT];
	inline ll qry(int x) {
		ll ret=0;

			add(ret,tr[x]);
		}

inline ll ksm(ll x,ll y) {
	ll ans=1;

		y>>=1;
	}
	return ans;
}

#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int P=998244353;
const int MAXN=1e6+7;
int id,n,m;
ll f[MAXN],sum[MAXN];

int home[MAXN],nxt[MAXN],to[MAXN],ccnt,in[MAXN];

inline void ct(int x,int y) {
	ccnt++;
	nxt[ccnt]=home[x];
	home[x]=ccnt;
	to[ccnt]=y;
}

inline void add(ll &x,ll y) {
	x+=y;
	if(x>=P)x-=P;
}

int main() {
	scanf("%d%d%d",&n,&n,&m);
	for(int i=1,x,y; i<=m; ++i) {
		scanf("%d%d",&x,&y);
		ct(x,y);
		in[x]++;
	}
	for(int i=1; i<=n; ++i) {
		in[i]++;
	}
	//dp！
	ll ans=0;
	for(int u=1; u<=n; ++u) {
		f[u]=1;
		sum[u]=sum[u-1];
		if(in[u]==1) { //只有一步QAQ
			add(ans,1);
			add(sum[u],f[u]);
			continue;
		}
		for(int i=home[u]; i; i=nxt[i]) {
			int v=to[i];
			add(f[u],(sum[u]-sum[v-1]+1+P)%P);
		}
		add(sum[u],f[u]);
		add(ans,f[u]);
	}
	printf("%lld\n",ans);
	return 0;
}
