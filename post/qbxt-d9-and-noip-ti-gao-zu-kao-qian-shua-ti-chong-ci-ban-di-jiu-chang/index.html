//容斥
#include<bits/stdc++.h>
const int MAXN = 4e6 + 7;
#define ll long long
using namespace std;
const int P = 998244353;//摸错人
int n, N;
int fac[MAXN], ifac[MAXN];
ll cf6[MAXN];
inline ll ksm(ll x, ll y) {
	ll ans = 1;
	while(y) {
		if(y & 1)ans = ans * x % P;
		x = x * x % P;
		y >>= 1;
	}
	return ans;
}

inline void add(ll &x, ll y) {
	x += y;
	if(x >= P)x -= P;
}

inline void sub(ll &x, ll y) {
	x -= y;
	if(x < 0)x += P;
}

inline void init() {
	fac[0] = 1;
	ifac[0] = 1;
	for(int i = 1; i <= N; ++i)fac[i] = 1ll * fac[i - 1] * i % P;
	ifac[N] = ksm(fac[N], P - 2);
	for(int i = N - 1; i >= 1; --i)ifac[i] = 1ll * ifac[i + 1] * (i + 1) % P;
	cf6[0] = 1;
	for(int i = 1; i <= n; ++i)cf6[i] = cf6[i - 1] * 6 % P;
	return;
}

inline ll C(int n, int m) {
	return 1ll * fac[n] * ifac[m] % P * ifac[n - m] % P;
}

inline void solve() {
	ll ans = fac[N];
	for(int i = 1; i <= n; ++i) {
		if(i & 1)sub(ans, cf6[i] * C(n, i) % P * fac[N - 2 * i]  % P);
		else add(ans, cf6[i] * C(n, i) % P * fac[N - 2 * i]  % P);
	}
	printf("%lld\n", ans);
}

int main() {
	scanf("%d", &n);
	N = 3 * n;
	init();
	solve();
	return 0;
}
