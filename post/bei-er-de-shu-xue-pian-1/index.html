inline Cmpl ksmC(Cmpl x, ll y, ll P) {
	Cmpl ans = Cmpl(1, 0);
	while(y) {
		if(y & 1)ans = ans * x;
		x = x * x;
		y >>= 1;
	}
	return ans;

	return ksmC(Cmpl(x, 0), (P - 1) / 2, P).A == P - 1;
