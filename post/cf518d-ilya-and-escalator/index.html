直接DP,记f[i][j]表示第i秒有j个人的期望

转移直接转移就做完了

然后code:

#include<iostream>
#include<cstdio>
#include<cstring>
const int MAXN = 2e3 + 7;
int n, m;
double p;
double dp[MAXN][MAXN];

int main() {
	scanf("%d", &n);
	scanf("%lf", &p);
	scanf("%d", &m);
	for(int i = 1; i <= n; ++i) {
		for(int j = 1; j <= m; ++j) {
			dp[i][j] = (1 - p) * dp[i][j - 1] + p * (dp[i - 1][j - 1] + 1);
		}
	}
	printf("%lf\n", dp[n][m]);
	return 0;
}
