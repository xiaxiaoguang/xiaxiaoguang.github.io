#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int MAXN = 1e5 + 7;
const int P = 1e9 + 7;
int n, k, a[MAXN];
ll f[MAXN];

inline ll ksm(ll x, ll y) {
	ll ans = 1;
	while(y) {
		if(y & 1)ans = ans * x % P;
		x = x * x % P;
		y >>= 1;
	}
	return ans;
}

int main() {
	scanf("%d%d", &n, &k);
	for(int i = 1; i <= n; ++i) {
		scanf("%d", &a[i]);
		if(i == 1) {
			f[0] = a[i];
			f[1] = P - 1;
			continue;
		}
		for(int j = i; j ; --j) {
			f[j] = ((-f[j - 1] + f[j] * a[i] % P) % P + P) % P;
		}
		f[0] = 1ll * f[0] * a[i] % P;
	}
	ll ans = 0;
	for(int i = 0; i <= n; ++i) {
		ll tmp = f[i];
		// printf("%lld\n", f[i]);
		for(int j = k - i + 1; j <= k; ++j) {
			tmp = tmp * j % P;
		}
		tmp = tmp * ksm(ksm(n, i), P - 2) % P;
		// printf("%lld\n", tmp);
		ans += tmp;
		ans %= P;
	}
	printf("%lld\n", (f[0] - ans + P) % P);
	return 0;
}
