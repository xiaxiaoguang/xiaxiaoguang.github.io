//让世界计数吧!
#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int P = 998244353;
const int MAXN = 4050;
int n, K, l;
inline int add(int x, int y) {
	x += y;
	if(x >= P)x -= P;
	return x;
}

inline ll ksm(ll x, ll y) {
	ll ans = 1;
	while(y) {
		if(y & 1)ans = ans * x % P;
		x = x * x % P;
		y >>= 1;
	}
	return ans;
}

ll fac[MAXN], ifac[MAXN];

inline void init() {
	fac[0] = ifac[0] = 1;
	for(int i = 1; i <= 2 * n + 1; ++i)fac[i] = fac[i - 1] * i % P;
	ifac[2 * n + 1] = ksm(fac[2 * n + 1], P - 2);
	for(int i = 2 * n; i >= 1; --i)ifac[i] = ifac[i + 1] * (i + 1) % P;
}

int f[MAXN][MAXN][2];

int main() {
	scanf("%d%d%d", &n, &K, &l);
	init();
	f[0][0][0] = 1;
	for(int i = 1; i <= 2 * n + 1; ++i) {
		for(int j = 0; j <= i; ++j) {
			for(int k = 0; k <= 1; ++k) {
				f[i][j][k] = add(f[i][j][k], 1ll * (j + 1) * f[i - 1][j + 1][k] % P);
				f[i][j][k] = add(f[i][j][k], f[i - 1][j - 1][k]);
				if(k && j >= K)
					f[i][j][k] = add(f[i][j][k], f[i - 1][j][k ^ 1]);
			}
		}
	}
	printf("%lld\n", 1ll * f[2 * n + 1][0][1] * ksm(2, n) % P * fac[n] % P * ifac[2 * n + 1] % P * l % P);
	return 0;
}
